PROG     START   1000
         LDA     #0
         STA     COUNT
         LDX     #0
INPUT    TD      INDEV
         JEQ     INPUT
         LDA     #0
         RD      INDEV
         COMP    EOF
         JEQ     REV
         STCH    STR,X
         LDA     COUNT
         ADD     #1
         STA     COUNT
         LDX     COUNT
         J       INPUT
REV      LDA     COUNT
         COMP    #0
         JEQ     DONE
         LDA     COUNT
         SUB     #1
         STA     COUNT
         LDX     COUNT
OUTPUT   TD      OUTDEV
         JEQ     OUTPUT
         LDCH    STR,X
         WD      OUTDEV
         LDA     COUNT
         SUB     #1
         COMP    #0
         JLT     DONE
         STA     COUNT
         LDX     COUNT
         J       OUTPUT
DONE     J       *

OUTDEV   BYTE    X'05'
INDEV    BYTE    X'F3'
STR      RESB    100
COUNT    RESW    1
EOF      BYTE    X'000004'
         END
